Modify sample values to what you use.
```xml
<?xml version="1.0" encoding="utf-8" ?>
<configuration>
  <appSettings>
    <add key="RunningProcessName" value="MUST_RUN"/>
    <add key="WatchProcessName" value="TOOL_TO_LAUNCH"/>
    <add key="LaunchPath" value="C:\PATH\TO\TOOL_TO_LAUNCH.exe"/>
  </appSettings>
</configuration>
```
e.g.
```xml
<?xml version="1.0" encoding="utf-8" ?>
<configuration>
  <appSettings>
    <add key="RunningProcessName" value="Diablo III"/>
    <add key="WatchProcessName" value="Turbo"/>
    <add key="LaunchPath" value="C:\PATH\TO\Turbo.exe"/>
  </appSettings>
</configuration>
```
